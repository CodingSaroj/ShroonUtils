227  void * SUTL_InternalHashsetInsert(SUTLHashset * hs)
228  {
229  /*
230  * Calculate the index of the key stored in `ParamK`.
231  */
232  size_t index = hs->Hash(hs->ParamK) % SUTL_HASHSET_BUCKET_COUNT;
233 
234  hs->Size++;
235 
236  void * entry = NULL;
237 
238  size_t i;
239 
240  /*
241  * Check if the entry with key stored in `ParamK` already exists.
242  */
243  for (i = 0; i < SUTLVectorSize(hs->Keys[index]) / hs->KeySize; i++)
244  if (hs->KeyComp(hs->ParamK, hs->Keys[index] + i * hs->KeySize))
245  /*
246  * Initialize `entry` to the entry that already exists.
247  */
248  entry = hs->Keys[index] + i * hs->KeySize;
249 
250  /*
251  * If the entry already exists return it, otherwise proceed to add the new entry.
252  */
253  if (entry)
254  return entry;
255 
256  /*
257  * Push the new key at the end of the bucket with index `index` and return it.
258  */
259  return SUTLVectorPushN(hs->Keys[index], hs->ParamK, hs->KeySize);
260  }
