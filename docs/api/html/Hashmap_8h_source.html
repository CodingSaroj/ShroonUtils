259  void * SUTL_InternalHashmapInsert(SUTLHashmap * hm)
260  {
261  /*
262  * Calculate the index of the key stored in `ParamK`.
263  */
264  size_t index = hm->Hash(hm->ParamK) % SUTL_HASHMAP_BUCKET_COUNT;
265 
266  hm->Size++;
267 
268  void * value = NULL;
269 
270  size_t i;
271 
272  /*
273  * Check if the entry with key stored in `ParamK` already exists.
274  */
275  for (i = 0; i < SUTLVectorSize(hm->Keys[index]) / hm->KeySize; i++)
276  if (hm->KeyComp(hm->ParamK, hm->Keys[index] + i * hm->KeySize))
277  /*
278  * Initialize `value` to the value of the entry that already exists.
279  */
280  value = hm->Values[index] + i * hm->KeySize;
281 
282  /*
283  * If the entry already exists return it, otherwise proceed to add the new entry.
284  */
285  if (value)
286  return value;
287 
288  /*
289  * Push the new key at the end of the bucket with index `index`.
290  */
291  SUTLVectorPushN(hm->Keys[index], hm->ParamK, hm->KeySize);
292 
293  /*
294  * Push the new value at the end of the bucket with index `index` and return only the
295  * value.
296  */
297  return SUTLVectorPushN(hm->Values[index], hm->ParamV, hm->ValueSize);
298  }

301  {
302  /*

304  */
305  size_t index = hm->Hash(hm->ParamK) % SUTL_HASHMAP_BUCKET_COUNT;
306 
307  size_t i;
308 
309  /*
310  * Search the bucket with index `index` for the key stored in `ParamK`.
311  */
312  for (i = 0; i < SUTLVectorSize(hm->Keys[index]); i++)
313  {
314  if (hm->KeyComp(hm->ParamK, hm->Keys[index] + i * hm->KeySize))
315  {

318  */

320  SUTLVectorEraseN(hm->Values[index], i * hm->ValueSize, hm->ValueSize);
