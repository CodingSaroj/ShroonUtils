212  void * SUTL_InternalVectorNew(size_t elemsize)
213  {

240  void SUTL_InternalVectorReserve(void ** v, size_t size)
241  {
242  /*
243  * Can't reserve memory less than the memory assigned to elements. This can cause loss of
244  * data.
245  *
246  * However, note that this would work if `size` is less than both the capacity and the size
247  * of the vector. This allows the extra memory to be shrunk down to the size of the used
248  * memory.
249  */
250  if (size < SUTLVectorSize(*v))
251  {
252  SUTLErrorHandler("Can't reserve memory less than the amount of memory already in use.");
253  return;
254  }
255 
256  size_t requiredSize =
257  sizeof(size_t) * 3 /* For variables size, capacity and element size. */
258  + size * SUTLVectorElemsize(*v) /* For elements allocated in the vector. */
259  ;
260 
261  /*
262  * Resize the memory of vector (including the internal variables) to the `requiredSize`.
263  */
264  *v = SHRN_REALLOC((size_t *)*v - 3, requiredSize);
265 
266  if (!*v)
267  {
268  SUTLErrorHandler("Memory reallocation failed.");
269  return;
270  }
271 
272  /*
273  * Make sure `v` points to the start of the array instead of internal variables.:
274  */
275  *v = (size_t *)*v + 3;
276 
277  SUTLVectorCapacity(*v) = size;
278  }

280  void SUTL_InternalVectorResize(void ** v, size_t size)
281  {
282  /*
283  * Reserve memory only if current memory is not more than required memory.
284  */
285  if (SUTLVectorSize(*v) < size)
286  if (SUTLVectorCapacity(*v) < size)
287  /*
288  * For now, it reserves size for 2 extra elements.
289  *
290  * This behaviour might change later.
291  */
292  SUTL_InternalVectorReserve(v, size + 2);
293 
294  /*
295  * Change the size anyway.
296  *
297  * If current size is less than required size than uninitialized elements are added.
298  * If current size is same as required size than nothing happens.
299  * If current size is greater than required size than extra elements become "free space".
300  */
301  SUTLVectorSize(*v) = size;
302  }

304  void * SUTL_InternalVectorInsertN(void ** v, size_t at, const void * ptr, size_t count)
305  {
306  /*
307  * `at` becomes the index of the first added element hence it must be less than or equal to
308  * the original size of vector.
309  */
310  if (at > SUTLVectorSize(*v))
311  {
312  SUTLErrorHandler("Insert index must be less than or equal to size.");
313  return NULL;
314  }
315 
316  size_t originalSize = SUTLVectorSize(*v);
317 
318  /*
319  * Resize the vector to the sum of current size and count. From here, the size of vector
320  * is the new size.
321  */
322  SUTL_InternalVectorResize(v, originalSize + count);
323 
324  /*
325  * No need to shift elements if `at` is the end of vector.
326  */
327  if (at != originalSize)
328  /*
329  * Shift elements from `at` to the right by `count` to make space for the new element.
330  */
331  SHRN_MEMMOVE(
332  SUTLVectorOffset(*v, at + count),
333  SUTLVectorOffset(*v, at),
334  (originalSize - at) * SUTLVectorElemsize(*v)
335  );
336 
337  /*
338  * Set the contents of the inserted elements to be the contents at `ptr` and return the
339  * pointer to the first inserted element.
340  */
341  return SHRN_MEMCPY(SUTLVectorOffset(*v, at), ptr, SUTLVectorElemsize(*v) * count);
342  }

344  void SUTL_InternalVectorEraseN(void ** v, size_t at, size_t count)
345  {
346  /*
347  * There must be `count` elements available from index `at`.
348  */
349  if (at + count > SUTLVectorSize(*v))
350  {
351  if ((at > SUTLVectorSize(*v)))
352  {
353  SUTLErrorHandler("Elements requested to be erased don't exist.");
354  return;
355  }
356 
357  count = SUTLVectorSize(*v) - at;
358  }
359 
360  /*
361  * No need to shift elements if the elements to be removed are at the end of vector.
362  */
363  if (at + count != SUTLVectorSize(*v))
364  /*
365  * Shift elements from `at` to the left by `count` to overwrite the elements to erase
366  * with the data of elements after it.
367  */
368  SHRN_MEMMOVE(
369  SUTLVectorOffset(*v, at),
370  SUTLVectorOffset(*v, at + count),
371  (SUTLVectorSize(*v) - count - at) * SUTLVectorElemsize(*v)
372  );
373 
374  SHRN_MEMSET(
375  SUTLVectorOffset(*v, SUTLVectorSize(*v) - count),
376  0,
377  SUTLVectorElemsize(*v) * count
378  );
379 
380  SUTLVectorSize(*v) -= count;
381  }
